def ricker(t, s=1.0):
    a = 2 /(np.sqrt(3 * s) * np.pi**0.25)
    b = 1 - (t / s)**2
    c = np.e**(-1 * t**2 / (2 * s**2))
    return a * b * c

s = 0.1
x = np.arange(-10, 10, s)
y1 = ricker(x,1.0)
plt.plot(x, y1)
plt.show()

def cwt(d, mw, wid):
    y = np.zeros([len(wid), len(d)])
    for i, j in enumerate(wid):
        wave = mw(min(10 * j, len(d)), j)
        y[i, :] = np.convolve(d, wave, mode='same')
    return y
